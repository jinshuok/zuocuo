/* 节气精确计算（天文算法） */
const JIEQI_NAMES = [
  '小寒','大寒','立春','雨水','惊蛰','春分',
  '清明','谷雨','立夏','小满','芒种','夏至',
  '小暑','大暑','立秋','处暑','白露','秋分',
  '寒露','霜降','立冬','小雪','大雪','冬至'
];

function solarTermJD(year, n) {
  const angles = [285,300,315,330,345,0,15,30,45,60,75,90,105,120,135,150,165,180,195,210,225,240,255,270];
  const angle = angles[n];
  const k = (year - 2000) * 365.25 / 365.25;
  let jd = 2451545.0 + (angle / 360) * 365.25 + k * 365.25;

  for (let i = 0; i < 50; i++) {
    const lambda = sunLongitude(jd);
    let diff = angle - lambda;
    if (diff > 180) diff -= 360;
    if (diff < -180) diff += 360;
    const dt = diff / 360 * 365.25;
    jd += dt;
    if (Math.abs(dt) < 0.00001) break;
  }
  return jd;
}

function sunLongitude(jd) {
  const T = (jd - 2451545.0) / 36525.0;
  const L0 = 280.46646 + 36000.76983 * T + 0.0003032 * T * T;
  const M = (357.52911 + 35999.05029 * T - 0.0001537 * T * T) * Math.PI / 180;
  const C = (1.914602 - 0.004817 * T - 0.000014 * T * T) * Math.sin(M)
           + (0.019993 - 0.000101 * T) * Math.sin(2 * M)
           + 0.000289 * Math.sin(3 * M);
  const sunLon = L0 + C;
  const omega = (125.04 - 1934.136 * T) * Math.PI / 180;
  const lambda = sunLon - 0.00569 - 0.00478 * Math.sin(omega);
  return ((lambda % 360) + 360) % 360;
}

function jdToDate(jd) {
  const ms = (jd - 2440587.5) * 86400000;
  return new Date(ms);
}

function nextJieqiDate() {
  const now = new Date();
  const jdNow = 2440587.5 + now.getTime() / 86400000;
  const year = now.getFullYear();

  let best = null;
  let bestName = '';
  for (let y = year; y <= year + 1; y++) {
    for (let n = 0; n < 24; n++) {
      const jd = solarTermJD(y, n);
      if (jd > jdNow + 0.001) {
        if (!best || jd < best) {
          best = jd;
          bestName = JIEQI_NAMES[n];
        }
      }
    }
  }
  return { date: jdToDate(best), name: bestName };
}

function calcReminderTime(type) {
  const now = new Date();
  switch (type) {
    case '15m': return new Date(now.getTime() + 15 * 60 * 1000);
    case '1d':  return new Date(now.getTime() + 24 * 60 * 60 * 1000);
    case '1w':  return new Date(now.getTime() + 7 * 24 * 60 * 60 * 1000);
    case '1y':  return new Date(now.getTime() + 365 * 24 * 60 * 60 * 1000);
    case 'jieqi': return nextJieqiDate().date;
    default: return null;
  }
}

function formatReminderPreview(type) {
  const d = calcReminderTime(type);
  if (!d) return '';
  if (type === 'jieqi') {
    const { name } = nextJieqiDate();
    return '将在「' + name + '」提醒：' + fmtDatetime(d);
  }
  return '提醒时间：' + fmtDatetime(d);
}

function fmtDatetime(d) {
  return d.getFullYear() + '-' + p2(d.getMonth()+1) + '-' + p2(d.getDate()) + ' ' + p2(d.getHours()) + ':' + p2(d.getMinutes());
}
function p2(n) { return String(n).padStart(2,'0'); }

function scheduleNotif(title, body, at) {
  const delay = at.getTime() - Date.now();
  if (delay <= 0) return;
  if (delay > 2147483647) {
    storeScheduledNotif(title, body, at);
    return;
  }
  setTimeout(() => {
    if (Notification.permission === 'granted') {
      new Notification(title, { body, icon: '/favicon.ico' });
    }
  }, delay);
  storeScheduledNotif(title, body, at);
}

function storeScheduledNotif(title, body, at) {
  const key = 'buzuo_notifs';
  const existing = JSON.parse(localStorage.getItem(key) || '[]');
  existing.push({ title, body, at: at.toISOString() });
  localStorage.setItem(key, JSON.stringify(existing));
}

function checkStoredNotifs() {
  const key = 'buzuo_notifs';
  const existing = JSON.parse(localStorage.getItem(key) || '[]');
  const now = Date.now();
  const remaining = [];
  existing.forEach(n => {
    const at = new Date(n.at).getTime();
    const delay = at - now;
    if (delay <= 0) {
      if (now - at < 3600000 && Notification.permission === 'granted') {
        new Notification(n.title, { body: n.body, icon: '/favicon.ico' });
      }
    } else if (delay <= 2147483647) {
      setTimeout(() => {
        if (Notification.permission === 'granted') {
          new Notification(n.title, { body: n.body, icon: '/favicon.ico' });
        }
      }, delay);
      remaining.push(n);
    } else {
      remaining.push(n);
    }
  });
  localStorage.setItem(key, JSON.stringify(remaining));
}

}

/* History Modal */
async function openHist(taskId) {
  histTaskId = taskId;
  const t = tasks.find(x => x.id===taskId); if (!t) return;
  g('histChip').textContent = t.text;
  g('histNote').value = '';
  await loadHist(taskId);
  openM('histModal');
}

async function loadHist(taskId) {
  const { data, error } = await sb.from('task_history').select('*').eq('task_id', taskId).order('created_at', { ascending: false });
  const el = g('histList');
  if (error || !data || !data.length) { el.innerHTML = '<div class="no-hist">还没有记录</div>'; return; }
  el.innerHTML = '';
  data.forEach(r => {
    const div = document.createElement('div');
    div.className = 'hist-row';
    const d = new Date(r.created_at);
    div.innerHTML = '<div class="hist-dot"></div><div><div class="hist-date">' + fmtDatetime(d) + '</div>' + (r.note?'<div class="hist-note">' + esc(r.note) + '</div>':'') + '</div>';
    el.appendChild(div);
  });
}

async function addHist() {
  if (!histTaskId) return;
  const note = g('histNote').value.trim();
  const { error } = await sb.from('task_history').insert({ task_id: histTaskId, user_id: user.id, note: note||null });
  if (error) { toast(friendlyErr(error), true); return; }
  const t = tasks.find(x => x.id===histTaskId);
  const newCnt = (t?.undo_count||0) + 1;
  await sb.from('tasks').update({ undo_count: newCnt, updated_at: new Date().toISOString() }).eq('id', histTaskId);
  if (t) t.undo_count = newCnt;
  g('histNote').value = '';
  await loadHist(histTaskId);
  render(); toast('已记录这次不做');
}

/* @ Mention Modal */
function openMention(taskId) {
  mentionTaskId = taskId; mentionRemVal = null;
  const t = tasks.find(x => x.id===taskId); if (!t) return;
  g('mentionTaskChip').textContent = t.text;
  g('mentionTaskChip').className = 'chip ' + (t.task_type==='love' ? 'chip-love' : 'chip-quit');
  g('mentionTarget').value = '';
  g('mentionTimePreview').textContent = '';
  document.querySelectorAll('#mentionRemGrid .mention-rem-chip').forEach(b => b.classList.remove('on'));
  openM('mentionModal');
}

g('mentionRemGrid').addEventListener('click', e => {
  const b = e.target.closest('.mention-rem-chip'); if (!b) return;
  document.querySelectorAll('#mentionRemGrid .mention-rem-chip').forEach(x => x.classList.remove('on'));
  b.classList.add('on'); mentionRemVal = b.dataset.val;
  g('mentionTimePreview').textContent = formatReminderPreview(mentionRemVal);
});

async function sendMention() {
  if (!mentionTaskId) return;
  const toUser = g('mentionTarget').value.trim().toLowerCase();
  if (!toUser) { toast('@用户名不能为空', true); return; }
  if (!mentionRemVal) { toast('请选择过期时间', true); return; }
  if (toUser === username) { toast('不能 @ 自己', true); return; }

  const t = tasks.find(x => x.id===mentionTaskId); if (!t) return;
  const expiresAt = calcReminderTime(mentionRemVal);

  const { data: profileData } = await sb.from('profiles').select('user_id').eq('username', toUser).single();
  const toUserId = profileData?.user_id || null;

  const { error } = await sb.from('mentions').insert({
    from_user_id: user.id,
    to_username: toUser,
    to_user_id: toUserId,
    task_id: mentionTaskId,
    task_text: t.text,
    task_type: t.task_type,
    from_username: username,
    status: 'pending',
    expires_at: expiresAt.toISOString()
  });
  if (error) { toast(friendlyErr(error), true); return; }
  closeM('mentionModal');
  toast('已 @ ' + toUser + '，过期于 ' + fmtDatetime(expiresAt));
}

/* Inbox */
async function loadInbox() {
  if (!user) return;
  const now = new Date().toISOString();
  const { data, error } = await sb.from('mentions')
    .select('*')
    .eq('to_user_id', user.id)
    .eq('status', 'pending')
    .gt('expires_at', now)
    .order('created_at', { ascending: false });
  if (error) { toast(friendlyErr(error), true); return; }
  mentions_inbox = data || [];
  renderInbox();
}

function renderInbox() {
  const el = g('inboxList');
  el.innerHTML = '';
  g('inboxEmpty').style.display = mentions_inbox.length ? 'none' : '';
  mentions_inbox.forEach(m => {
    const div = document.createElement('div');
    div.className = 'inbox-item';
    const expiry = new Date(m.expires_at);
    div.innerHTML = 
      '<div class="inbox-from">@来自 ' + esc(m.from_username) + '</div>' +
      '<div class="inbox-text">' + (m.task_type==='love'?'爱做':'不做') + '：' + esc(m.task_text) + '</div>' +
      '<div class="inbox-meta">过期：' + fmtDatetime(expiry) + '</div>' +
      '<div class="inbox-actions">' +
        '<button class="act love" onclick="openReply(\'' + m.id + '\',\'love\')">加入爱做</button>' +
        '<button class="act quit" onclick="openReply(\'' + m.id + '\',\'quit\')">加入不做</button>' +
        '<button class="act del" onclick="ignoreMention(\'' + m.id + '\')">忽略</button>' +
      '</div>';
    el.appendChild(div);
  });
  updateInboxBadge();
}

function updateInboxBadge() {
  const cnt = mentions_inbox.length;
  const badge = g('inboxBadge');
  const dot = g('notifDot');
  if (cnt > 0) {
    badge.textContent = cnt; badge.style.display = '';
    dot.classList.add('on');
  } else {
    badge.style.display = 'none';
    dot.classList.remove('on');
  }
}

function clearInboxBadge() {
  g('notifDot').classList.remove('on');
}

function openReply(mentionId, defaultType) {
  replyMentionId = mentionId;
  replyType = defaultType;
  const m = mentions_inbox.find(x => x.id===mentionId); if (!m) return;
  g('replyChip').textContent = '来自 @' + m.from_username + '：' + m.task_text;
  setReplyType(defaultType);
  openM('replyModal');
}

function setReplyType(t) {
  replyType = t;
  g('replyLoveBtn').className = 'recv-type-btn' + (t==='love' ? ' love' : '');
  g('replyQuitBtn').className = 'recv-type-btn' + (t==='quit' ? ' quit' : '');
}

async function acceptMention() {
  if (!replyMentionId) return;
  const m = mentions_inbox.find(x => x.id===replyMentionId); if (!m) return;

  const { data: newTask, error: taskErr } = await sb.from('tasks').insert({
    user_id: user.id, text: m.task_text,
    status: 'plan', task_type: replyType, undo_count: 0
  }).select().single();
  if (taskErr) { toast(friendlyErr(taskErr), true); return; }

  const newStatus = replyType==='love' ? 'accepted_love' : 'accepted_quit';
  await sb.from('mentions').update({ status: newStatus, responded_at: new Date().toISOString() }).eq('id', replyMentionId);

  tasks.push({ ...newTask, createdAt: Date.now(), updatedAt: Date.now() });
  mentions_inbox = mentions_inbox.filter(x => x.id !== replyMentionId);
  closeM('replyModal'); render(); renderInbox();
  toast('已加入' + (replyType==='love'?'爱做':'不做') + '，对方将收到通知');
}

async function ignoreMention(id) {
  await sb.from('mentions').update({ status: 'ignored', responded_at: new Date().toISOString() }).eq('id', id);
  mentions_inbox = mentions_inbox.filter(x => x.id !== id);
  renderInbox();
  toast('已忽略');
}

/* Sent */
async function loadSent() {
  if (!user) return;
  const { data, error } = await sb.from('mentions').select('*').eq('from_user_id', user.id).order('created_at', { ascending: false });
  if (error) { toast(friendlyErr(error), true); return; }
  mentions_sent = data || [];
  renderSent();
}

function renderSent() {
  const el = g('sentList');
  el.innerHTML = '';
  g('sentEmpty').style.display = mentions_sent.length ? 'none' : '';
  const now = new Date();
  mentions_sent.forEach(m => {
    const div = document.createElement('div');
    div.className = 'inbox-sent-item';
    const expired = new Date(m.expires_at) < now;
    let statusLabel = '待处理';
    let statusClass = 'pending';
    if (m.status === 'accepted_love') { statusLabel = '对方加入了爱做'; statusClass = 'accepted'; }
    else if (m.status === 'accepted_quit') { statusLabel = '对方加入了不做'; statusClass = 'accepted'; }
    else if (m.status === 'ignored') { statusLabel = '对方已忽略'; statusClass = 'ignored'; }
    else if (expired) { statusLabel = '已过期'; statusClass = 'ignored'; }
    div.innerHTML = 
      '<div class="sent-to">→ @' + esc(m.to_username) + ' <span class="sent-status ' + statusClass + '">' + statusLabel + '</span></div>' +
      '<div class="tc-text" style="margin:10px 0">' + (m.task_type==='love'?'爱做':'不做') + '：' + esc(m.task_text) + '</div>' +
      '<div style="font-size:13px;color:var(--text-tertiary)">发送于 ' + fmtDatetime(new Date(m.created_at)) + ' · 过期 ' + fmtDatetime(new Date(m.expires_at)) + '</div>';
    el.appendChild(div);
  });
}

/* Realtime */
function setupRealtime() {
  if (realtimeChan) { sb.removeChannel(realtimeChan); }
  realtimeChan = sb.channel('mentions-' + user.id)
    .on('postgres_changes', {
      event: 'INSERT', schema: 'public', table: 'mentions',
      filter: 'to_user_id=eq.' + user.id
    }, payload => {
      const m = payload.new;
      mentions_inbox.unshift(m);
      renderInbox();
      if (Notification.permission === 'granted') {
        new Notification('收到 @ 邀请', { body: m.from_username + ' 分享了：' + m.task_text });
      } else {
        toast(m.from_username + ' @ 了你：' + m.task_text);
      }
    })
    .on('postgres_changes', {
      event: 'UPDATE', schema: 'public', table: 'mentions',
      filter: 'from_user_id=eq.' + user.id
    }, payload => {
      const m = payload.new;
      const idx = mentions_sent.findIndex(x => x.id===m.id);
      if (idx >= 0) mentions_sent[idx] = m;
      const t = tasks.find(x => x.id===m.task_id);
      if (t && (m.status==='accepted_love'||m.status==='accepted_quit')) {
        t._mention_accepted = true;
        render();
        if (Notification.permission === 'granted') {
          new Notification('@ 已被接受', { body: m.to_username + ' 接受了你分享的「' + m.task_text + '」' });
        } else {
          toast(m.to_username + ' 接受了你的 @ 分享');
        }
      }
    })
    .subscribe();
}

/* Auth State */
sb.auth.onAuthStateChange(async (_e, session) => {
  user = session?.user || null;
  if (user) {
    username = user.email.split('@')[0];
    g('uBadge').textContent = '@' + username;
    showPage('main');
    switchNav('home');
    await requestNotifPermission();
    checkStoredNotifs();
    loadTasks();
    loadInbox();
    setupRealtime();
  } else {
    tasks = []; mentions_inbox = []; mentions_sent = [];
    if (realtimeChan) sb.removeChannel(realtimeChan);
    showPage('auth');
  }
});

/* Expiry cleanup */
setInterval(() => {
  const now = new Date();
  const before = mentions_inbox.length;
  mentions_inbox = mentions_inbox.filter(m => new Date(m.expires_at) > now);
  if (mentions_inbox.length !== before) renderInbox();
}, 5 * 60 * 1000);
